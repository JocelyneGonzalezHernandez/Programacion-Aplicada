void agrega(struct nodo **raz, char caract[100]){
                struct nodo *av,*new;
                int r=0;
                av=*raz;
                if ((new=(struct nodo*)malloc(sizeof(struct nodo)))==NULL){
                        printf("No hay memoria\n");
                        exit(1);
                        }
                strcpy(new->l,caract);
                new->freq=1;
                new->izq=NULL;
                new->der=NULL;

                while(av !=NULL && r==0){
                        if(strcmp(caract,av->l)>0){
                                if(strcmp(caract,av->l)!=0){
                                        if (av->der!=NULL){
                                                        av=av->der;
                                                }
                                                else{
                                                        av->der=new;
                                                        return;
                                                }}
                                                else{
                                                r=1;
                                                av->freq=av->freq+1;
                                                }
                        }
                        if(strcmp(caract,av->l)<=0){
                                if(strcmp(caract,av->l)!=0){
                                        if (av->izq!=NULL){
                                                        av=av->izq;
                                                }
                                                else{
                                                        av->izq=new;
                                                        return;
                                                }
                                        }
                                                else{
                                                r=1;
                                                av->freq=av->freq+1;
                                                }
                }

                }
                if(r==0){
                        av=new;
                        *raz=av;
                }
}
int imprimir(struct nodo *t){
        if(t!=NULL){
                imprimir(t->izq);
                printf("%s aparece %d veces \n",t->l,t->freq);
                imprimir(t->der);
        }

}

int busca(struct nodo *t,char pal[100],int c){

        if(t!=NULL){
                busca(t->izq,pal,++c);
                if (strcmp(t->l,++pal)==0){
                        printf("%s aparece %d veces en la posicion %d\n",t->l,t->freq,c);
                        return 0;
                }
                if (t->izq!=NULL || t->der!=NULL){
                        ++c;
                }
                busca(t->der,pal,c++);
        }

}

int borra(struct nodo *t){
        if(t!=NULL){
                borra(t->izq);
                borra(t->der);
                free(t);
        }
}
