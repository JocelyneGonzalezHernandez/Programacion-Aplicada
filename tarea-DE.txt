#include"local.h"

void crearlista(lista **inicio,lista **aux,elementos contenido)
{
  lista *nodo;
  nodo=malloc(sizeof(lista));
  if(nodo==NULL)
    {
      printf("No hay memoria\n");
      exit(1);
    }
  nodo->contenido=contenido;

  if(*inicio==NULL)
    {
      *inicio=nodo;
      nodo->ant=NULL;
    }
  else
    {
      nodo->ant=*aux;
      (*aux)->sig=nodo;
    }
  *aux=nodo;
  nodo->sig=NULL;
}
void guardainfoBin(char *argv[],lista *inicio,elementos contenido)
{
  lista *aux;
  FILE *archivo;
  archivo=fopen(argv[2],"wb");
  aux=inicio;
  while(aux!=NULL)
    {
      fwrite(&aux->contenido,sizeof(elementos),1,archivo);
      aux=aux->sig;
    }
  fclose(archivo);
}
void liberalista(lista *inicio)
{
  lista *nodo;
  nodo=inicio;
  while(nodo != NULL)
    {
      inicio=nodo->sig;
      free(nodo);
      nodo=inicio;
    }
}
