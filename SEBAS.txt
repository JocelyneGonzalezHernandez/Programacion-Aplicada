//#define F(t) (pow((((3*(t))-1)/(((t)*(t))+3)),2))
#define F(t) (2*pow((t), 5))

double dPro(double x, double h)
{
  float res;
  res=((F(x+h))-(F(x)))/(h);
  return res;
}

double dCen(double x, double h)
{
  float res;
  res=((F(x+h))-(F(x-h)))/(2*h);
  return res;
}

double dReg(double x, double h)
{
  float res;
  res=((F(x))-(F(x-h)))/(h);
  return res;
}


int main()
{
  double a=2;
  double h= 0.001;

  cout<<"Progresiva\t x=2: "<<dPro(a,h)<<endl;
  cout<<"Regresiva\t x=2: "<<dReg(a,h)<<endl;
  cout<<"Cenral   \t x=2: "<<dCen(a,h)<<endl;

  return 0;
}

#include<iostream>
#include<cmath>
using namespace std;

//#define F(t) (pow((((3*(t))-1)/(((t)*(t))+3)),2))
#define F(t) (2*pow((t), 5))


double dPro(double x, double h)
{
  float res;
  res=((F(x+h+h))-(2*(F(x+h)))+(F(x)))/(h*h);
  return res;
}

double dCen(double x, double h)
{
  float res;
  res=((F(x+h+h))-(2*(F(x)))+(F(x-h-h)))/(4*h*h);
  return res;
}

double dReg(double x, double h)
{
  float res;
  res=((F(x))-(2*(F(x-h)))+(F(x-h-h)))/(h*h);
  return res;
}


int main()
{
  double a=2;
  double h= 0.001;
  cout<<"Progresiva\t x=2: "<<dPro(a,h)<<endl;
  cout<<"Regresiva\t x=2: "<<dReg(a,h)<<endl;
  cout<<"Cenral   \t x=2: "<<dCen(a,h)<<endl;

  return 0;
}

double f(double x, double h, int der)
{
  double res;
  if(der==1)
    {
      res=((F(x+h))-(F(x-h)))/(2*h);
      return res;
    }
  else
    {
      res=(f(x+h, h, der-1)-f(x-h, h, der-1))/(2*h);
      return res;
    }
}

  cout<<"D fy: "<<fx(x, y)<<endl;
  return 0;
}

double fx(double x, double y)
{
  double res;
  res=F(x+h, y);
  res=res-(F(x-h, y));
  res=res/(2*h);
  return res;
}
double fy(double x, double y)
{
  double res;
  res=F(x, y+h);
  res=res-(F(x, y-h));
  res=res/(2*h);
  return res;
}

double F(double x, double y)
{
  return ((x*x)+(x*y)+(y*y));
}
