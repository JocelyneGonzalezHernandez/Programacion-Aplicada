GtkWidget *AgregarBoton(GtkWidget *theBox, const gchar *buttonText, gpointer CallBackFunction, GtkWidget *EntryBox)
{
  GtkWidget *button;
  button = gtk_button_new_with_label(buttonText);
  gtk_box_pack_start(GTK_BOX(theBox),button,FALSE,TRUE,0);
  gtk_signal_connect(GTK_OBJECT(button),"clicked",GTK_SIGNAL_FUNC(CallBackFunction),EntryBox);
  gtk_widget_show(button);
  return button;
}

/**********************************************************************************************
 *
 * Funcion ClickResultado
 * Esta función realiza los cuatro tipos e operacion(suma, resta, multilicacion, division),
 * acorde a lo que el usuario teclee
 *
 *********************************************************************************************/
void ClickResultado(GtkButton *button, gpointer data)
{
  Nodo *datos=(Nodo *)data;
  float Numero2, Resultado;
  int i,Num1, Num2;
  float exp;
  char CadenaResultado [20], TempEntrada [20];
  const gchar *Entrada;
  FILE *fp;
  time_t t;
  struct tm *tm;
  char fecha_hora[50];
  Entrada = gtk_entry_get_text (GTK_ENTRY (data));
  strcpy (TempEntrada, Entrada);
  Numero2 = atof (TempEntrada);
  switch (datos -> Operacion [0])
  {
    case '+':
    Resultado = datos -> Numero + Numero2;
    break;
    case '-':
    Resultado = datos -> Numero - Numero2;
    break;
    case '/':
    Resultado = datos -> Numero / Numero2;
    break;
    case '*':
    Resultado = datos -> Numero * Numero2;
    break;
    case '!':
      Resultado=1;
      for(i=1; i<=datos->Numero; ++i)
      {
        Resultado = Resultado*i;
      }
    break;
    case 'P':
      Resultado=1;
      for(exp=Numero2; exp>0; exp--)
      {
        Resultado=Resultado*(datos->Numero);
      }
    break;
  }
  sprintf (CadenaResultado, "%.2f", Resultado);
  gtk_entry_set_text (GTK_ENTRY (data), CadenaResultado);
        fp=fopen("operaciones.txt", "a");

  t=time(NULL);
  tm=localtime(&t);
  strftime(fecha_hora, 50, "%d/%m/%Y-%I:%M:%S", tm);

        fprintf(fp,"\n%f %s %f = %.3f %s", datos->Numero, datos->Operacion, Numero2, Resultado,fecha_hora);
        fclose(fp);
}

/******************************************************************************
 *
 * Funcion ClickM_mas
 * Esta funcion establece que el boton M+ sume el numero almacenado
 * en la memoria
 *
 *****************************************************************************/
void ClickM_mas(GtkButton *button, gpointer data)
{
    Nodo *datos=(Nodo *)data;
    const gchar *original;
    gchar nueva[100], lectura[100], operacion[100];
    gfloat res,org,resultado;
    original = gtk_entry_get_text(GTK_ENTRY(data));
    strcpy(nueva, original);
    FILE *fp;
    fp=fopen("memoria.txt","r");
    if(fp==NULL){
        memoria(nueva);
    }
    else{
        fgets(lectura,100,fp);
        res=atof(lectura);
        org=atof(original);
        resultado=res+org;
        sprintf(lectura,"%f",resultado);
        sprintf(operacion, "%f+%f",res,org);
        gtk_entry_set_text(GTK_ENTRY(data),lectura);
        memoria(lectura);
        fclose(fp);
    }
    return;
}

{

    gfloat res,org,resultado;
    original = gtk_entry_get_text(GTK_ENTRY(data));
    strcpy(nueva, original);
    FILE *fp;
    fp=fopen("memoria.txt","r");
    if(fp==NULL){
        memoria(nueva);
    }
    else{
        fgets(lectura,100,fp);
        res=atof(lectura);

    }
    return;
}

#include "local.h"
/*****************************************************************************
 *
 * Funcion abrir
 * Abre el archivo de texto con operacione registradas en una nueva ventana
 * Recibe: widget y user_data
 * Regresa: no regresa ningún valor
 *
 ****************************************************************************/
void abrir(GtkWidget *widget, gpointer user_data)
{
    GtkWidget *ventanaRegistro,*etiqueta1,*vBox;

    ventanaRegistro = gtk_window_new(GTK_WINDOW_TOPLEVEL);
    vBox = gtk_vbox_new(TRUE, 2);
    FILE *fp;
    char cadena[100];
    fp= fopen("operaciones.txt","r");
    while(fgets(cadena,100,fp)!=NULL){
        etiqueta1 = gtk_label_new(cadena);
        gtk_window_set_title(GTK_WINDOW(ventanaRegistro), "Registro de Operaciones");
        gtk_widget_set_usize(ventanaRegistro, 400, 400);
        gtk_box_pack_start_defaults(GTK_BOX(vBox), etiqueta1);
    }
    fclose(fp);
    gtk_container_add(GTK_CONTAINER(ventanaRegistro), vBox);
    gtk_widget_show_all(ventanaRegistro);
}
/********************************************************************************
 *
 * Funcion cerrar
 * Cierra el archivo de texto con operacione registradas en una nueva ventana
 * Recibe: widget y user_data
 * Regresa: no regresa ningún valor
 *
 *******************************************************************************/
void cerrar(GtkWidget *widget, gpointer user_data)
{
    FILE *fp;
    fp=fopen("operaciones.txt", "r");
    fclose(fp);
}
