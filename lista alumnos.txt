void Inserta(tipoalumno **Inicio, char nombre[],float cal1,float cal2,float cal3)
{
  tipoalumno *temp;
  temp=(tipoalumno *)malloc(sizeof(tipoalumno)); //crea el espacion de memoria
  strcpy((*temp).nombre,nombre);//copia nombre a donde apunta temp con direccion de nombre
  (*temp).cal1=cal1;
  (*temp).cal2=cal2;
  (*temp).cal3=cal3;
  (*temp).prom=0;
  *Inicio=temp;  //donde apunta inicio sera temp
}
void Imprime(tipoalumno *Inicio)
{
  tipoalumno *temp;
  temp=Inicio;
  while(temp != NULL)
    {
      puts((*temp).nombre);
      printf("%.2f,%.2f,%.2f promedio: %.2f",(*temp).cal1,(*temp).cal2,(*temp).cal3,(*temp).prom);
      temp=(*temp).sig;
    }
}
void borra(tipoalumno **Inicio)
{
  tipoalumno *temp;
  temp=*Inicio;
  while(temp!=NULL)
    {
      *Inicio=(*temp).sig;
      free(temp);
      temp=*Inicio;
    }
}
