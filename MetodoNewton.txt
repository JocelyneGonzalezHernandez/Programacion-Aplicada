float df(float ta, float h) //calcula la derivada de la función F utilizando una aproximación por diferencias 
{
    float t = ta + h;
    return ((F(t)) - (F(ta))) / h; //Usando el Método de Newton
}

int Compara(float iteraciones[], int i) //Compara las raíces para ver las últimas son iguales
