extern void imprimir (DATOS **inicio, DATOS *nodo, DATOS *aux); //Funcion de imprimir trabajo, modulo 2
extern void cola_impresion (DATOS *inicio,DATOS *aux);  //Funcion de ver la cola de impresion, modulo 2

#include "estructura.h"

void agregar_trabajo (DATOS **inicio,DATOS *nodo, DATOS **aux, char *nombre, int *n_hojas){  //Funcion para agregar la informacion de un trabajo a la lista FIFO
  FILE *fp;
  fp=fopen ("Archivos.txt","at");
  //Pedir la informacion
  printf ("Ingrese el nombre del archivo: ");
  scanf ("%s",nombre);
  printf ("Ingrese el numero de hojas del archivo: ");
  scanf ("%d",n_hojas);
  fprintf (fp,"%s \t\t",nombre);  //Guardar en el archivo el nombre del tabajo
  fprintf (fp,"%d \n",*n_hojas);  //Guardar en el archivo el numero de hojas del trabajo
  fclose (fp);  //Cierra el archivo
  //Crear el espacio
  nodo=malloc (sizeof (DATOS));
  //Llenar la informacion
  strcpy (nodo->nombre,nombre);
  nodo->n_hojas=(*n_hojas);
  //Encadenar
  if (*inicio==NULL){
    *inicio=nodo;
  }
  else{
    (*aux)->sig=nodo;
  }
  (*aux)=nodo;
  nodo->sig=NULL;

}//Fin de la funcion agregar_trabajo


void liberar (DATOS *inicio, DATOS *nodo){  //Funcion para liberar el espacio al salir del programa
  nodo=inicio;
  while (nodo != NULL){
    inicio=inicio->sig;
    free (nodo);
    nodo=inicio;
  }
}//Fin de la funcion liberar

int menu (void){  //Menu principal con el que el usuario podra elejir si agregar o imprimir un trabajo, ademas de ver la cola de impresion o salir del programa
  int opc;
  printf ("**********************************\n");

  printf ("5.Salir.\n");

  printf ("Elija una de las opciones anteriores: ");
  scanf ("%d",&opc);
  return opc;
}

void imprimir (DATOS **inicio, DATOS *nodo, DATOS *aux){  //Funcion para imprimir el primer trabajo pendiente (primer trabajo de la lista FIFO)
  //Abre el archivo para guardar la informacion del trabajo
  aux=(*inicio);
  (*inicio)=aux->sig;
  free (aux);  //Elimina el primer trabajo de la lista FIFO
  aux=(*inicio);
  printf ("Trabajo impreso correctamente.\n");
}//Fin de la funcion imprimir

void cola_impresion (DATOS *inicio,DATOS *aux){  //Funcion para mostrar todos los trabajos que estan pendientes de imprimirse (trabajos que estan en la lista FIFO)
  printf("\n \n");
  printf ("\t La cola de impresion es la siguiente:\n");
  aux=inicio;
  printf ("Nombre del archivo \t");
  printf ("Hojas\n");
  while (aux != NULL){
    printf ("%s \t\t",aux->nombre);  //Imprimir el nombre del tabajo
    printf ("%d\n",aux->n_hojas);  //Imprimir el numero de hojas del trabajo
    aux=aux->sig;
  }
}//Fin de la funcion cola_impresion
