void botones11(GtkWidget *widget,gpointer info,Teclas *valores, float res)
{
  Teclas *valor = (Teclas *)info;
  const gchar *operacion;
  float n1,n2,resultado=1;
  int x;
  FILE *archivo;
  FILE *archivo2;
  char signo,cantidad[50],fecha[50];
  time_t tiempo=time(NULL);
  struct tm *localt = localtime(&tiempo);

  
  operacion = gtk_entry_get_text(GTK_ENTRY(valor->entry));

  sscanf(operacion,"%f %c %f",&n1,&signo,&n2);

  if(signo == '+')
    resultado = n1+n2;
  else if(signo == '-')
    resultado = n1-n2;
  else if(signo == '/')
    resultado = n1/n2;
  else if(signo == '*')
    resultado = n1*n2;
  else if(signo == 'P'){
    
    for(x = 0; x < n2; x++){
      resultado = n1*resultado;
    }
    
  }
  sprintf(cantidad,"%0.2f",resultado);
  gtk_entry_set_text(GTK_ENTRY(valor->entry),cantidad);
  strftime(fecha,50,"%d/%m/%y %H:%M:%S\n",localt);

  archivo = fopen("Historial.txt", "a");
  fprintf(archivo,"Fecha de la operacion: %s",fecha);
  fprintf(archivo,"Operacion: %0.2f %c %0.2f\n",n1,signo,n2);
  fprintf(archivo,"Resultado: %0.2f\n",resultado);

  archivo2 = fopen("Memoria.txt", "a");
  fprintf(archivo2,"%0.2f\n", resultado);
  fclose(archivo);
  fclose(archivo2);
}

void abrir(GtkWidget *widget, gpointer info,Teclas *valores)
{
  Teclas *valor=(Teclas *)info;
  FILE *archivo;
  char mensaje[50];
  GtkWidget *ver,*vertical,*hori,*boton,*ventana;
  GtkTextBuffer *contenido;
  archivo = fopen("Historial.txt", "r");
  if(archivo != NULL)
    {
      valor->window = gtk_dialog_new();
      vertical = gtk_dialog_get_content_area(GTK_DIALOG(valor->window));
      
      hori = gtk_dialog_get_action_area(GTK_DIALOG(valor->window));
      
      boton = gtk_button_new_with_label("Okey");
      gtk_window_set_title(GTK_WINDOW(valor->window),"Historial.txt");
      contenido = gtk_text_buffer_new(NULL);
      
     
      while(fgets(mensaje,50,archivo) != NULL)
	gtk_text_buffer_insert_at_cursor(GTK_TEXT_BUFFER(contenido),mensaje,strlen(mensaje));
      fclose(archivo);
      
      ver = gtk_text_view_new_with_buffer(contenido);
      
      gtk_text_view_set_editable(GTK_TEXT_VIEW(ver), FALSE);
      
      g_signal_connect_swapped(G_OBJECT(boton), "clicked",G_CALLBACK(gtk_widget_destroy),valor->window);
      
      gtk_box_pack_start_defaults(GTK_BOX(vertical), ver);
      gtk_box_pack_start_defaults(GTK_BOX(hori), boton);
      
      gtk_widget_show_all(valor->window);
    }
  else
    {
       ventana = gtk_dialog_new();
       vertical = gtk_dialog_get_content_area(GTK_DIALOG(ventana));
       hori = gtk_dialog_get_action_area(GTK_DIALOG(ventana));
       boton = gtk_button_new_with_label("close");
       contenido = gtk_text_buffer_new(NULL);
       
       gtk_text_buffer_set_text(contenido,"No existe ningun archivo",24);

       ver = gtk_text_view_new_with_buffer(contenido);
       
       gtk_text_view_set_editable(GTK_TEXT_VIEW(ver), FALSE);
       
       g_signal_connect_swapped(G_OBJECT(boton), "clicked",G_CALLBACK(gtk_widget_destroy), ventana);
       
       gtk_box_pack_start_defaults(GTK_BOX(vertical), ver);
       gtk_box_pack_start_defaults(GTK_BOX(hori), boton);
       
       gtk_widget_show_all(ventana);
