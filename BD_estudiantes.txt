CREATE TABLE ej1_alumnos(
idAlumno INT NOT NULL PRIMARY KEY AUTOINCREMENT,
nombre VARCHAR(25) NOT NULL,
genero ENUM('F', 'M') NOT NULL,
fecha DATE NOT NULL
);

CREATE TABLE ej1_ausencias(
idAusencias INT NOT NULL PRIMARY KEY AUTOINCREMENT,
fecha DATE NOT NULL,
idAlumno INT REFERENCES ej1_alumnos(idAlumno) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE ej1_materias(
idMateria INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
nombre VARCHAR(25) NOT NULL
);

CREATE TABLE ej1_tipoExamen(
idTipo INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
tipo ENUM('S', 'M', 'F') NOT NULL
);

CREATE TABLE ej1_examenes(
idExamen INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
fecha DATE NOT NULL,
idMateria INT REFERENCES ej1_materias(idMateria) ON DELETE CASCADE ON UPDATE CASCADE,
idTipo INT REFRENCES ej1_tipoExamen(idTipo)  ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE ej1_calificaciones(
idCalif INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
idAlumno INT REFERENCES ej1_alumnos(idAlumno) ON DELETE CASCADE ON UPDATE CASCADE,
idExamen INT REFERENCES ej1_examenes(idExamen) ON DELETE CASCADE ON UPDATE CASCADE,
calif NUMERIC (4, 2) NOT NULL
);

CREATE TABLE ej1_alumnos_materias(
idAlumnoMateria INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
idAlumno INT REFERENCES ej1_alumnos(idAlumno) ON DELETE CASCADE ON UPDATE CASCADE,
idMateria INT REFERENCES ej1_materias(idMateria) ON DELETE CASCADE ON UPDATE CASCADE,
semestre INT NOT NULL, 
status ENUM('C', 'B', 'A', 'R') NOT NULL
);
