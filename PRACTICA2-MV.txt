/********************************************************************
 *
 * Función  crearListaAcompanante
 * Descripción: Agrega acompañante(s) a un conductor, en caso
 *              de que el conductor no tenga espacios disponibles
 *              o no exista, no añade acompañante
 * Recibe: inicioCon nodoAcom, ,inicioAcom, auxAcom,nodoCom
 * Regresa: no regresa ningœn valor
 *
 *********************************************************************/
void crearListaAcompanante(CONDUCTORES *inicioCon)
{
    ACOMPANANTE *nodoAcom, *inicioAcom, *auxAcom;
    CONDUCTORES *nodoCon;
    char nombreCon[30];
    int encontrado=0;

    nodoCon=inicioCon;

    nodoAcom=malloc(sizeof(ACOMPANANTE));
    if(nodoAcom == NULL){
        printf("No hay memoria\n");
        exit(1);
    }

    printf("Ingrese el nombre del conductor al que desea agregar un acompañante\n");
    fflush(stdin);
    getchar();
    scanf("%[^\n]s", nombreCon);

    while(nodoCon!=NULL)
    {
        if(strcmp(nodoCon->nombre, nombreCon)==0)
        {
            if(nodoCon->lugares!=0)//Verifica si el conductor aún tiene lugares disponibles
            {
                encontrado=1;
                printf("¿Cuál es el nombre del acompañante?\n");
                fflush(stdin);
                getchar();
                scanf("%[^\n]s", nodoAcom->nombre);
                fflush(stdin);
                getchar();
                printf("¿Qué carrera estudia?\n");
                scanf("%[^\n]s", nodoAcom->carrera);
                fflush(stdin);
                getchar();
                printf("¿Cuál es su dirección?\n");
                scanf("%[^\n]s", nodoAcom->direccion);
                fflush(stdin);
                getchar();

                if(nodoCon->inicio == NULL){
                    nodoCon->inicio=nodoAcom;
                }
                else{
                    auxAcom=nodoCon->inicio;
                    inicioAcom=auxAcom;

                    while(auxAcom->ab!=NULL)
                    {
                        auxAcom=auxAcom->ab;
                    }
                    inicioAcom=auxAcom;
                    inicioAcom->ab=nodoAcom;
                }
            nodoAcom->ab = NULL;
            nodoCon->lugares=(nodoCon->lugares)-1;//Se le resta un lugares al número de lugares disponibles
            }
            else{
                printf("Este conductor no tiene lugares disponibles\n");
                encontrado=1;
            }
        }
        nodoCon=nodoCon->sig;
    }
    if(encontrado!=1)
    {
        printf("No se encontró ningún conductor con ese nombre\n");
    }
}


/********************************************************************
 *
 * Función   leerArchivoAcompanante
 * Descripción: Lee los datos de acompañantes desde un archivo
 *              binaro
 * Recibe: inicioCon,nodoCon,nodoAcom, inicioAcom, auxAcom
 * Regresa: no regresa ningœn valor
 *
 *********************************************************************/
void leerArchivoAcompanante(CONDUCTORES *inicioCon)
{
    CONDUCTORES *nodoCon;
    ACOMPANANTE *nodoAcom, *inicioAcom, *auxAcom;
    FILE *fp;
    int lugares;

    nodoCon=inicioCon;
    fp = fopen("datosAcompanantes.bin", "rb");
    if(fp != NULL){//Verifica si existe un archivo con el registro de acompañantes
                  //En caso de que el archvio tenga datos, se agregarán a las listas simples de acompañantes
                nodoAcom = malloc(sizeof(ACOMPANANTE));
        if(nodoAcom == NULL){
            printf("No hay memoria\n");
            exit(1);
        }

        while(nodoCon!=NULL)
        {
            lugares=(nodoCon->lugaresTot)-(nodoCon->lugares);//Con esta variable se sabe cuántos acompañantes se van a encadenar a cada conductor
            while(lugares!=0)//Cuando sea 0, significa que no se tiene que encadenar ningún nodo
            {
                fread(nodoAcom, sizeof(ACOMPANANTE), 1, fp);
                if(nodoCon->inicio == NULL){
                    nodoCon->inicio=nodoAcom;
                }
                else
                {
                    auxAcom=nodoCon->inicio;
                    while(auxAcom->ab!=NULL)
                    {
                        auxAcom=auxAcom->ab;
                    }
                    inicioAcom=auxAcom;
                    inicioAcom->ab=nodoAcom;
                }

                nodoAcom->ab = NULL;
                nodoAcom=malloc(sizeof(ACOMPANANTE));
                if(nodoAcom==NULL)
                {
                    printf("No hay memoria\n");
                    exit(1);
                }
                lugares=lugares-1;
            }
            nodoCon=nodoCon->sig;
        }
        fclose(fp);
    }
}

/********************************************************************
 *
 * Función  guardarArchivoAcompanante
 * Descripción: Guarda los datos de acompañantes de la lista
 *              a el archivo binario
 * Recibe: inicioCon , nodoCon ,nodoAcom, inicioAcom
 * Regresa: no regresa ningœn valor
 *
 *********************************************************************/
void guardarArchivoAcompanante(CONDUCTORES *inicioCon)
{
    CONDUCTORES *nodoCon;
    ACOMPANANTE *nodoAcom, *inicioAcom;
    FILE *fp;

    fp = fopen("datosAcompanantes.bin", "wb");

    nodoCon = inicioCon;
    while(nodoCon != NULL){
        if(nodoCon->inicio!=NULL)
        {
            inicioAcom=nodoCon->inicio;
            nodoAcom=inicioAcom;
            while(nodoAcom!=NULL)
            {
            fwrite(nodoAcom, sizeof(ACOMPANANTE), 1, fp);
            nodoAcom=nodoAcom->ab;
            }
        }
        nodoCon = nodoCon->sig;
    }

    fclose(fp);
}

/********************************************************************
 *
 * Funcion   borrarAcompanante
 * Descripcion :
 * Recibe: inicioCon,nodoCon,nodoAcom, inicioAcom, auxAcom
 * Regresa: no regresa ningún valor
 *
 *********************************************************************/
void borrarAcompanante(CONDUCTORES *inicioCon)
{
    CONDUCTORES *nodoCon;
    ACOMPANANTE *nodoAcom, *inicioAcom, *auxAcom;
    char nombreCon[30];
    char nombreAcom[30];
    int encontradoCon=0;
    int encontradoAcom=0;
    int borrado=0;

    nodoCon=inicioCon;

    printf("¿Cuál es nombre del conductor con quien se encuentra el acompañante?\n");
    fflush(stdin);
    getchar();
    scanf("%[^\n]s", nombreCon);

    while(nodoCon != NULL)
    {
        if(strcmp(nodoCon->nombre, nombreCon)==0)//Busca el nombre del conductor
        {
            encontradoCon=1;
            if(nodoCon->inicio!=NULL)
            {
                inicioAcom=nodoCon->inicio;
                nodoAcom=inicioAcom;
                printf("¿Cuál es el nombre del acompañante que desea borrar?\n");
                fflush(stdin);
                getchar();
                scanf("%[^\n]s", nombreAcom);

                if(strcmp(nodoAcom->nombre, nombreAcom)==0)//Entrará sólo si el acompañante que se desea borrar es el primero de la lista
                {
                    encontradoAcom=1;
                    inicioAcom=inicioAcom->ab;
                    nodoCon->inicio=inicioAcom;
                    free(nodoAcom);
                    borrado=1;
                    nodoCon->lugares=(nodoCon->lugares+1);
                }
                else
                {
                    while(nodoAcom!=NULL)
                    {
                        if(strcmp(nodoAcom->nombre, nombreAcom)==0)
                        {
                            encontradoAcom=1;
                            while(inicioAcom->ab!=nodoAcom)
                            {
                                inicioAcom=inicioAcom->ab;
                            }
                            auxAcom=nodoAcom->ab;
                            inicioAcom->ab=auxAcom;
                            free(nodoAcom);
                            borrado=1;
                            nodoCon->lugares=(nodoCon->lugares+1);
                        }
                        nodoAcom=nodoAcom->ab;
                    }
                }
            }
            else
            {
                printf("Este conductor no tiene acompañantes\n");
            }
        }
        nodoCon = nodoCon->sig;
    }
    if(encontradoCon!=1)
    {
        printf("No se encontró ningún conductor con ese nombre\n");
    }
    else
    {
        if(encontradoAcom!=1)
        {
            printf("No se encontró ningún acompañante con ese nombre\n");
        }
        if(borrado==1);
        {
            printf("Se ha borrado al acompañante\n");
        }
    }
}

 *
 * Función leerArchivoConductores
 * Descripción: Obtiene los datos de los conductores del archivo
 *                       binario y los importa a una lista.
 * Recibe: inicio ,aux, nodo
 * Regresa: no regresa ningœn valor
 *
 *********************************************************************/
void leerArchivoConductores(CONDUCTORES **inicio, CONDUCTORES **aux)
{
    CONDUCTORES *nodo;
    FILE *fp;

    fp = fopen("datosConductores.bin", "rb");
    if(fp != NULL){  //Verifica si existe un archivo con el registro de conductores
                    //En caso de que el archvio tenga datos, se agregarán a la lista doblemente encadenada
        nodo = malloc(sizeof(CONDUCTORES));
        if(nodo == NULL){
            printf("No hay memoria\n");
            exit(1);
        }

        while(fread(nodo, sizeof(CONDUCTORES), 1, fp) != NULL){

            if(*inicio == NULL){
                *inicio = nodo;
                nodo-> ant = NULL;
            }
            else{
                (*aux)->sig = nodo;
                nodo->ant = *aux;
            }
            *aux = nodo;
            nodo->sig = NULL;
            nodo->inicio=NULL;

            nodo = malloc(sizeof(CONDUCTORES));
            if(nodo == NULL){
                printf("No hay memoria\n");
                exit(1);
            }
        }

        fclose(fp);
    }
}

/********************************************************************
 *
 * Función   borrarConductor
 * Descripción: Elimina un conductor de la lista en caso de que exista
 * Recibe: inicio ,aux
 * Regresa: no regresa ningún valor
 *
 *********************************************************************/
void borrarConductor(CONDUCTORES **inicio, CONDUCTORES **aux)
{
    char nombreCon[30];
    int encontrado=0;
    CONDUCTORES *nodo;
    ACOMPANANTE *nodoAcom, *inicioAcom;

    nodo=*inicio;

    printf("Nombre del conductor que desea borrar:\n");
    fflush(stdin);
    getchar();
    scanf("%[^\n]s", nombreCon);

    if(strcmp((*inicio)->nombre, nombreCon)==0)//Sólo se entrará en caso de que el conductor que se quiera borrar sea el primero de la lista
    {
        encontrado=1;
        if((*inicio)->inicio!=NULL)//Verifica si el conductor tiene acompañantes
        {
            inicioAcom=nodo->inicio;
            nodoAcom=inicioAcom;
            while(nodoAcom!=NULL)
            {
                inicioAcom=nodoAcom->ab;
                free(nodoAcom);
                nodoAcom=inicioAcom;
            }
        }
        *inicio=nodo->sig;
        free(nodo);
    }

    else//En caso de que el conductor que se quiera borrar no sea el primero de la lista
    {
        while(nodo != NULL)
        {
            if(nodo==*aux)
            {
                if(strcmp(nodo->nombre, nombreCon)==0)
                {
                    if(nodo->inicio!=NULL)
                    {
                        inicioAcom=nodo->inicio;
                        nodoAcom=inicioAcom;

                        while(nodoAcom!=NULL)
                        {
                            inicioAcom=nodoAcom->ab;
                            free(nodoAcom);
                            nodoAcom=inicioAcom;
                        }
                    }
                    *aux=nodo->ant;
                    (*aux)->sig=NULL;
                    free(nodo);
                    encontrado=1;
                }
            }

            if(strcmp(nodo->nombre, nombreCon)==0)
            {
                if(nodo->inicio!=NULL)
                {
                    inicioAcom=nodo->inicio;
                    nodoAcom=inicioAcom;

                    while(nodoAcom!=NULL)
                    {
                        inicioAcom=nodoAcom->ab;
                        free(nodoAcom);
                        nodoAcom=inicioAcom;
                    }
                }
                nodo->ant->sig=nodo->sig;
                nodo->sig->ant=nodo->ant;
                free(nodo);
                encontrado=1;
            }
            nodo = nodo->sig;
        }
    }
    if(encontrado!=1)
    {
        printf("No se encontró ningún conductor con ese nombre\n");
    }
    else
    {
        printf("Se ha borrado al conductor\n");
    }

}
